#include <metal_stdlib>
using namespace metal;
struct Edge { float x0; float y0; float x1; float y1; float dir; };
struct RenderParams { int width; int height; int edge_count; uint8_t r; uint8_t g; uint8_t b; uint8_t a; };
kernel void clear_canvas(device uint8_t* canvas [[buffer(0)]], constant RenderParams& params [[buffer(1)]], uint2 tid [[thread_position_in_grid]]) {
    if (tid.x >= (uint)params.width || tid.y >= (uint)params.height) return;
    int off = (tid.y * params.width + tid.x) * 4;
    canvas[off]   = params.r;
    canvas[off+1] = params.g;
    canvas[off+2] = params.b;
    canvas[off+3] = params.a;
}
kernel void rasterize_edges(device uint8_t* canvas [[buffer(0)]], device const Edge* edges [[buffer(1)]], constant RenderParams& params [[buffer(2)]], uint2 tid [[thread_position_in_grid]]) {
    if (tid.x >= (uint)params.width || tid.y >= (uint)params.height) return;
    float py = float(tid.y) + 0.5;
    float px = float(tid.x) + 0.5;
    float winding = 0.0;
    for (int i = 0; i < params.edge_count; i++) {
        float y0 = edges[i].y0;
        float y1 = edges[i].y1;
        if (py >= y0 && py < y1) {
            float t = (py - y0) / (y1 - y0);
            float ix = edges[i].x0 + t * (edges[i].x1 - edges[i].x0);
            if (ix <= px) { winding += edges[i].dir; }
        }
    }
    if (abs(winding) > 0.001) {
        int off = (tid.y * params.width + tid.x) * 4;
        uint8_t a = params.a;
        if (a == 255) {
            canvas[off] = params.r; canvas[off+1] = params.g; canvas[off+2] = params.b; canvas[off+3] = a;
        } else if (a > 0) {
            int dst_r = canvas[off]; int dst_g = canvas[off+1]; int dst_b = canvas[off+2]; int dst_a = canvas[off+3];
            int inv_a = 255 - a;
            canvas[off] = (params.r + (dst_r * inv_a) / 255); canvas[off+1] = (params.g + (dst_g * inv_a) / 255);
            canvas[off+2] = (params.b + (dst_b * inv_a) / 255); canvas[off+3] = (a + (dst_a * inv_a) / 255);
        }
    }
}
